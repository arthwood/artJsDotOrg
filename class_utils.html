ClassUtils

Package:

com.arthwood.utils

Description

Facilitates class inheritance.

Methods

  • extend(base:Function, constr:Function, instanceMethods:Object, classMethods:Object):Function

    Creates a subclass ob base class

    Example:

    function Animal(name) {
      this.name = name;
    }

    Animal.prototype = {
      getSound: function() {
        return '...';
      }
    };

    Animal.getName = function() {
      return 'animal';
    };

    Dog = ClassUtils.extend(Animal, function(name, options) {
      arguments.callee.$super(name);

      this.options = options;
    }, {
      getSound: function() {
        var sound = arguments.callee.$super();

        return sound + 'meoww';
      }
    }, {
      getName: function() {
        var name = arguments.callee.$super();

        return name + '::' + 'dog';
      }
    });

    var dog = new Dog('fluffy', {color: 'white'});

    dog.name; // 'fluffy'
    dog.getSound(); // '...meoww'
    Dog.getName(); // 'animal::dog'
